Service audioplayer.service anlegen in

/etc/systemd/system

[Unit]
Description=Audio Player

[Service]
ExecStart=/usr/bin/node /home/pi/mh_prog/AudioServer/server.js > /home/pi/mh_prog/output-server.txt

[Install]
WantedBy=sysinit.target

[Unit]
Description=qtdemo
DefaultDependencies=no
[Service]
ExecStart=/usr/bin/qtdemoE -qws
[Install]
WantedBy=basic.target

in Autostart mit
systemctl enable audioplayer.service
